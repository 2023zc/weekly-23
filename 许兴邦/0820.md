```cpp
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
const int N = 1e6+10;
int e[N],ne[N];
int idx,head;
void insert(int k,int x){
    e[idx]=x;
    ne[idx]=ne[k];
    ne[k]=idx;
    idx++;
}
void del(int k){
    ne[k]=ne[ne[k]];
}
void hea(int x){
    e[idx]=x;
    ne[idx]=head;
    head=idx;
    idx++;
}
void init(){
    idx=0;
    head=-1;
}
int main()
{
    int n;
    scanf("%d", &n);
    init();
    while (n -- ){
        char ch;
        cin>>ch;
        if(ch=='H'){
            int x;
            cin>>x;
            hea(x);
        }
        else if(ch=='I'){
            int k,x;
            cin>>k>>x;
            insert(k-1,x);
        }
        else if(ch=='D'){
            int k;
            cin>>k;
            if(k)
            del(k-1);
            else{
                head=ne[head];
            }
        }
    }
    for(int i=head;i!=-1;i=ne[i]){
        cout<<e[i]<<" ";
    }
    cout<<endl;
}
```

#include <iostream>
#include <cstring>
#include <algorithm>

const int N = 1e6+10;
int n,l[N],r[N],idx,e[N];
void ir(int k,int x){
    e[idx]=x;
    r[idx]=r[k];
    l[idx]=k;
    l[r[k]]=idx;
    r[k]=idx;
    idx++;
}
void D(int k){
    r[l[k]]=r[k];
    l[r[k]]=l[k];
}

void init(){
    r[0]=1;
    l[1]=0;
    idx=2;
    
}
int main()
{
    // ios::sync_with_stdio(false);

            int k;
            cin>>k;
            D(k+1);

        }
}
